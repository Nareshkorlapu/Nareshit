Write a C program to insert an element at a specified position in an array ?
-----------------------------------------------------------------------------

Sample Input :-
---------------
Enter the number of elements in the array :- 5
Enter 5 elements :- 10 20 30 40 50
Enter the element to be inserted :- 25
Enter the position :- 3

Sample Output :-
-----------------
Array after insertion: 10 20 25 30 40 50

Sol:-
-------------------------------------
#include <stdio.h>

int main() {
    int n, element, position;

    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    int arr[n+1]; 

    printf("Enter %d elements: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter the element to be inserted: ");
    scanf("%d", &element);
    
    printf("Enter the position: ");
    scanf("%d", &position);

    for (int i = n; i >= position; i--) {
        arr[i] = arr[i-1];
    }

    arr[position - 1] = element;

    n++;

    printf("Array after insertion: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}

Write a C program to delete a given element from an array ?
------------------------------------------------------------

Sample Input :-
----------------
Enter the number of elements in the array: 8
Enter 8 elements: 10 20 30 30 30 40 50 30
Enter the element to be deleted: 30

Sample Output :-
----------------
Array after deletion: 10 20 40 50

Sol:-
-----------------------------------------------
#include <stdio.h>

int main() {
    int n, element;

    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter %d elements: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter the element to be deleted: ");
    scanf("%d", &element);

    int i, j;
    for (i = 0; i < n; i++) {
        if (arr[i] == element) {

            for (j = i; j < n - 1; j++) {
                arr[j] = arr[j + 1];
            }
            n--; 
            i--;  
        }
    }

    printf("Array after deletion: ");
    for (i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}

Write a C program to print all the prime numbers from an array ?
----------------------------------------------------------------

Sample Input :-
----------------
Enter the number of elements in the array: 7
Enter 7 elements: 12 5 9 7 11 2 8

Sample Output :-
-----------------
Prime numbers in the array: 5 7 11 2

Sol:-
------------------------
#include <stdio.h>

int main() {
    int n;
    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d elements: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Prime numbers in the array: ");
    for (int i = 0; i < n; i++) {
        int num = arr[i];
        if (num <= 1) {
            continue; 
        }
        int p = 1;  
        for (int j = 2; j <= num / 2; j++) {
            if (num % j == 0) {
                p = 0;  
                break;
            }
        }
        if (p) {
            printf("%d ", num);
        }
    }
    return 0;
}

Write a C program to print all the unique elements of an array ?
-----------------------------------------------------------------
Sample Input :-
----------------
Enter the number of elements in the array: 6
Enter 6 elements: 10 20 20 30 40 10

Sample Output :-
-----------------
Unique elements in the array: 30 40

Sol:-
--------------------------------
#include <stdio.h>

int main() {
    int n;
    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d elements: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Unique elements in the array: ");
    for (int i = 0; i < n; i++) {
        int uni = 1;
        for (int j = 0; j < n; j++) {
            if (i != j && arr[i] == arr[j]) {
                uni= 0;
                break;
            }
        }
        if (uni) {
            printf("%d ", arr[i]);
        }
    }
    return 0;
}
