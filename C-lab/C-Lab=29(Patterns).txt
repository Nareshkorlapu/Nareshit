|-------------|
| Program-1 : |
|-------------|

Write C program to given pattern.
---------------------------------
|-----------------|
| *  *  *  *  *   |
| *  *  *  *      |
| *  *  *         |
| *  *            |
| *               |
|-----------------|

Sol:-
----------
// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int n;
    printf("Enter n value:");
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        for(int j=n;j>=i;j--){
            printf("*  ");
        }
        printf("\n");
    }

    return 0;
}

|----------------|

Sol:-
---------

#include <stdio.h>

int main() {
    char ch,ch2;
    printf("Enter character:");
    scanf(" %c",&ch);
    for(char i='A';i<=ch;i++){
        ch2=ch;
        for(char j=ch;j>=i;j--){
            printf("%3c",ch2--);
            
        }
        printf("\n");
    }

    return 0;
}

| Program-4 : |
|-------------|

Write C program to print given pattern.
---------------------------------------

|----------------|
|             *  |
|          *  *  |
|       *     *  |
|    *        *  | 
| *  *  *  *  *  |
|----------------|

Sol:-
------------------
// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int n;
    printf("Enter n value:");
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        for(int j=n;j>i;j--){
            printf("   ");
        }
        for(int k=1;k<=i;k++){
            if(i==k || k==1 || i==n){
            printf("*  ");
            }
            else{
                printf("   ");
            }
        }
        printf("\n");
    }

    return 0;
}
